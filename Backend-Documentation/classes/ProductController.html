import { Request, Response } from "express";
import { IProduct, Product } from "../models/productModel";


export class ProductController {
  public async createProduct(req: Request, res: Response) {
    console.log("Create Product");
    await Product.findOne({ creator: req.user._id, productname: req.body.productname }, async (err: any, results: any) => {
      if (!err && results === null) {
        const product = new Product({
          ...req.body,
          creator: req.user._id,
        })
        try {
          await product.save()
          res.status(201).send({ product })
        } catch (e) {
          res.status(400).send(e)
        }
      } else {
        res.status(200).send("Product already exists")
      }
    })
  }

  public async getProducts(req: Request, res: Response) {
    console.log("Get Products")
    try {
      const products: IProduct[] = await Product.find(
        {},
        {},
        {
          limit: parseInt(String(req.query.limit)),
          skip: parseInt(String(req.query.skip))
        })
      res.status(200).send(products)
    } catch (e) {
      res.status(500).send()
    }
  }

  public async updateProduct(req: Request, res: Response) {
    const updates = Object.keys(req.body)
    const allowedUpdates = ['productname', 'price', 'description', 'image']
    const isValidOperation = updates.every((update) => allowedUpdates.includes(update))
    if (!isValidOperation) {
      return res.status(400).send({ error: 'Invalid updates!' })
    }
    try {
      if (req.body.role === "Admin") {
        Product.updateOne({ _id: req.query._id }, await req.body, { upsert: true }, async function (err, doc) {
          if (err) {
            console.log(err)
            return res.status(400).send({ error: err })
          }
          res.status(200).send()
        });
      }
      if (req.body.role === "Staff") {
        Product.updateOne({ creator: req.user._id, _id: req.query._id }, await req.body, { upsert: true }, async function (err, doc) {
          if (err) {
            console.log(err)
            return res.status(400).send({ error: err })
          }
          res.status(200).send()
        });
      }
    } catch (e) {
      res.status(400).send(e)
    }
  }

  public async deleteProduct(req: Request, res: Response) {
    console.log(req.query._id)
    try {
      if (req.body.role === "Admin") {
        Product.findOneAndDelete(
          {
            _id: req.query._id
          },
          {},
          function (err, docs) {
            if (err) {
              console.log(err)
              res.status(400).send(err)
            }
            else {
              console.log("Deleted User : ", docs);
              if (docs === null) {
                return res.status(204).send({ error: 'No product was deleted!' })
              }
              res.status(200).send(docs)
            }
          });
      }
      if (req.body.role === "Staff") {
        Product.findOneAndDelete(
          {
            creator: req.user._id,
            _id: req.query._id
          },
          {},
          function (err, docs) {
            if (err) {
              console.log(err)
              res.status(400).send(err)
            }
            else {
              console.log("Deleted User : ", docs);
              if (docs === null) {
                return res.status(204).send({ error: 'No product was deleted!' })
              }
              res.status(200).send(docs)
            }
          });
      }
    } catch (e) {
      console.log(e);
      res.status(400).send(e)
    }
  }

  public async getProductList(req: Request, res: Response) {
    console.log(req.user)
    console.log("Get Products")
    const sort: { [key: string]: any } = {}
    if (req.query.sortBy) {
      const parts = String(req.query.sortBy).split(':')
      sort[parts[0]] = parts[1] === 'desc' ? -1 : 1
    }
    try {
      if (req.user.role === "Staff") {
        console.log("staff")
        const products: IProduct[] = await Product.find(
          { creator: req.user._id },
          {},
          {
            limit: parseInt(String(req.query.limit)),
            skip: parseInt(String(req.query.skip)),
            sort
          })
        res.status(200).send(products)
      }
      if (req.user.role === "Customer") {
        console.log("customer")
        const products: IProduct[] = await Product.find(
          { customer: req.user._id },
          {},
          {
            limit: parseInt(String(req.query.limit)),
            skip: parseInt(String(req.query.skip)),
            sort
          })
        res.status(200).send(products)
      }
    } catch (e) {
      res.status(500).send()
    }
  }
